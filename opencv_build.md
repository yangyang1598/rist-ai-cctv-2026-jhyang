```bash
# 가상환경 위치 설정
REPO_DIR="/home/${USER}/rist-ai-cctv-2025-2nd"
VENV_PATH="${REPO_DIR}/.venv"
PYTHON_VERSION="python3.12"
PYTHON_LIBRARY="/usr/lib/x86_64-linux-gnu/lib${PYTHON_VERSION}.so"
PYTHON_INCLUDE_DIR="/usr/include/${PYTHON_VERSION}"
PYTHON_EXECUTABLE="${VENV_PATH}/bin/python"
PYTHON_NUMPY_INCLUDE_DIR="${VENV_PATH}/lib/${PYTHON_VERSION}/site-packages/numpy/core/include"
PYTHON_PACKAGES_PATH="${VENV_PATH}/lib/${PYTHON_VERSION}/site-packages"
NV_CODEC_SDK="${REPO_DIR}/src/Video_Codec_SDK_13.0.19"
CUDA_COMPUTE_POWER=8.9
CUDA_NVCUVID_LIBRARY="${NV_CODEC_SDK}/Lib/linux/stubs/x86_64/libnvcuvid.so"

```

```bash
git clone https://github.com/opencv/opencv
git clone https://github.com/opencv/opencv_contrib

cd opencv
mkdir build && cd build
```

```bash
cmake -D CMAKE_BUILD_TYPE=RELEASE -D CMAKE_INSTALL_PREFIX=/usr/local -D OPENCV_EXTRA_MODULES_PATH=${OPENCV_EXTRA_MODULES_PATH} -D INSTALL_PYTHON_EXAMPLES=ON -D INSTALL_C_EXAMPLES=ON -D BUILD_DOCS=OFF -D BUILD_TESTS=OFF -D ENABLE_FAST_MATH=ON -D CUDA_FAST_MATH=ON -D WITH_CUDA=ON -D WITH_CUBLAS=ON -D WITH_CUFFT=ON -D WITH_NVCUVID=ON -D WITH_NVCUVENC=ON -D WITH_CUDNN=ON -D OPENCV_DNN_CUDA=ON -D CUDA_ARCH_BIN=${CUDA_COMPUTE_POWER} -D CUDA_ARCH_PTX=${CUDA_COMPUTE_POWER} -D BUILD_opencv_cudacodec=ON -D WITH_GSTREAMER=ON -D WITH_FFMPEG=ON -D WITH_V4L=ON -D WITH_OPENGL=ON -D WITH_OPENCL=ON -D WITH_TBB=ON -D WITH_EIGEN=ON -D WITH_LAPACK=ON -D WITH_PROTOBUF=ON -D WITH_JPEG=ON -D WITH_PNG=ON -D WITH_TIFF=ON -D WITH_WEBP=ON -D WITH_GTK=ON -D WITH_QT=ON -D WITH_OPENEXR=ON -D OPENCV_ENABLE_NONFREE=ON -D BUILD_opencv_python3=ON -D BUILD_opencv_python2=OFF -D PYTHON3_EXECUTABLE=${PYTHON_EXECUTABLE} -D PYTHON3_INCLUDE_DIR=${PYTHON_INCLUDE_DIR} -D PYTHON3_NUMPY_INCLUDE_DIRS=${PYTHON_NUMPY_INCLUDE_DIR} -D PYTHON3_PACKAGES_PATH=${PYTHON_PACKAGES_PATH} -D PYTHON3_LIBRARY=${PYTHON_LIBRARY} -D CUDA_nvcuvid_LIBRARY=${CUDA_NVCUVID_LIBRARY} -D CUDA_nvidia-encode_LIBRARY=${CUDA_NVENC_LIBRARY} -D CUDA_CUDA_LIBRARY=${CUDA_CUDA_LIBRARY} ..
    
make -j$(nproc)
```
